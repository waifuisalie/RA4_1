(10 # 5)

(7 $ 2)

(3.14.15 X)

(..5 Y)

(10. A)

(20 & 15)

(8 %% 3)

# Caso 9: Número malformado com espaços
(5 . 5 X)

# Caso 10: Múltiplos caracteres inválidos
(@ # $ %)

((5 3 +)

(+)

# Caso 14: Operador multiplicação sem operandos
(*)

(5 + 3)

# Caso 16: Notação infixa com subtração
(10 - 5)

5 3 +

(+ 5 3)

(5 3 + -)

()

# Caso 21: Divisão inteira com operando real
(5.5 2 /)

# Caso 22: Operação resto com operando real
(10.5 3 %)

# Caso 23: Potenciação com expoente real (deve ser int)
(2 3.5 ^)

# Caso 24: Variável não inicializada
(UNDEFINED_VAR)

# Caso 25: Operador lógico AND com tipos numéricos
(5 3 &&)

# Caso 26: Operador lógico OR com tipos numéricos
(10 20 ||)

((5 3 >) X)
